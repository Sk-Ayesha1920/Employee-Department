5.select sum(sal) as total_salary from emp;
6.select sum(nvl(comm,0)) as total_commission from emp;
7.select distinct job from emp where comm is not null;
8.select sysdate as system_date from dual;
9.select avg(sal) as average_salary from emp;
10.select deptno, count(*) as employees_per_department from emp group by deptno;
11.select deptno, sum(sal) as total_salary_per_department from emp group by deptno;
12.select job, count(*) as employees_per_job from emp group by job;
13.select job, avg(sal) as average_salary_per_job from emp group by job;
14.select ename, extract(day from hiredate) as hire_day, extract(month from hiredate) as hire_month, extract(year from hiredate) as hire_year from emp;
15.select * from emp order by deptno;
16.select * from emp order by job;
17.select * from emp order by sal desc;
18.select * from emp order by deptno,sal desc;
19.select * from emp where length(ename)=6;
20.select max(sal) as max_salary, min(sal) as min_salary from emp;
21.select deptno, max(sal) as max_salary, min(sal) as min_salary, avg(sal) as avg_salary, sum(sal) as total_salary from emp group by deptno;
22.select * from emp order by hiredate;
23.select * from emp where hiredate=(select max(hiredate) from emp);
24.select * from emp where hiredate=(select min(hiredate) from emp);
25.select * from emp order by extract(year from hiredate) desc, deptno;
